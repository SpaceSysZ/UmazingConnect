# Microsoft Azure Authentication Configuration
# You need to register your app in the Azure Portal and get these values

# Your Azure App Registration Client ID
NEXT_PUBLIC_AZURE_CLIENT_ID=your-client-id-here

# Optional: Azure Tenant ID (if you want to restrict to specific tenant)
# NEXT_PUBLIC_AZURE_TENANT_ID=your-tenant-id-here

# Optional: Redirect URI (should match what you configure in Azure)
# NEXT_PUBLIC_REDIRECT_URI=http://localhost:3000

# Steps to set up Azure authentication:
# 1. Go to https://portal.azure.com
# 2. Navigate to "Azure Active Directory" > "App registrations"
# 3. Click "New registration"
# 4. Name: "SchoolConnect"
# 5. Supported account types: "Accounts in this organizational directory only"
# 6. Redirect URI: Web > http://localhost:3000
# 7. After creation, copy the "Application (client) ID"
# 8. Go to "Authentication" and add http://localhost:3000 as redirect URI
# 9. Go to "API permissions" and add Microsoft Graph > User.Read permission
# 10. Create a .env.local file with your client ID
